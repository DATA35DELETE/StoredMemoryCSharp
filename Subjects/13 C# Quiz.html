namespace storedMemoryCSharp
{
    internal class Program
    {
        static void Main(string[] args)
        {
            List<string> esyaListesi = new List<string>();

            esyaListesi.Add("Dolap");
            esyaListesi.Add("Masa");
            esyaListesi.Add("Sandalye");
            esyaListesi.Add("Halı");
            esyaListesi.Add("Yatak");

            Console.Write("Alınacak eşyalar: ");

            foreach (string esya in esyaListesi)
            {
                Console.Write(esya + ",  ");
            }
        }
    }
}
