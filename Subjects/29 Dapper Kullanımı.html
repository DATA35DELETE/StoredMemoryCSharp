Bağlantı Kurma ve Kütüphaneler

    Bağlantı kurmak için kütüphaneleri projemize dahil etmek gerekir. Bunun için;
using Dapper;
using MySql.Data.MySqlClient;
    Kütüphanelerini ekleyelim.

Bağlantı Kurma

    Bağlantı kurmak için ilk önce bağlantı sorgusu yazmamız gerek.
string conn = "Server=localhost;" + // Server IP'si
    "Database=database;" + // Bağlancak veritabını ismi
    "User ID=root;" + // Yetkilendirme
    "Password=password;"; // Şifre
    Şeklinde bir sorgu yazılmalıdır.
   Ardından;
using var connection = new MySqlConnection(conn); // Bağlantıyı gönder
connection.Open(); // Bağlantıyı aç
    ile de bağlantımız kurulmuş olur.

Sorgu Gönder ve Veri Alımı

var result = connection.Query("SHOW DATABASES;"); // SQL sorgusunu gönder ve gelen veriyi al
    Veriyi göstermek içinse;
foreach (var db in result)                           //
{                                                    // Sonuçları foreach ile 
    Console.WriteLine($"Veritabanı: {db.Database}"); //  döndürerek yazıdır
}
