        private void button4_Click(object sender, EventArgs e)
        {
            forkeredProjectName = ((checkBox1.Checked) ? "" : "SM") + textBox2.Text;

            if (forkeredProjectName != null && forkeredProjectName.Length > 0 && !forkeredProjectName.Equals("SM"))
            {
                FolderBrowserDialog folderDialog = new FolderBrowserDialog();

                folderDialog.Description = "Bir klasör seçin";
                folderDialog.ShowNewFolderButton = true;

                if (folderDialog.ShowDialog() == DialogResult.OK)
                {
                    string secilenKlasor = folderDialog.SelectedPath;
                    
                    label3.Text = secilenKlasor;

                    forkeredProjectPath = secilenKlasor;

                    tabControl1.SelectedIndex = 3;
                }
                else
                {
                    MessageBox.Show("A directory must be selected!");

                    tabControl1.SelectedIndex = 2;
                }
            }
            else
            {
                MessageBox.Show("Your information is incorrect or incomplete!");
            }
        }

        private void iisVideoUse_CheckedChanged(object sender, EventArgs e)
        {
            if (iisVideoUse.Checked)
            {
                videoFile = null;
                wallpaperImage = null;

                OpenFileDialog openFileDialog = new OpenFileDialog();

                openFileDialog.Filter = "Video Files (*.mp4)|*.mp4";
                openFileDialog.Title = "Select an video";

                if (openFileDialog.ShowDialog() == DialogResult.OK)
                {
                    string dosyaYolu = openFileDialog.FileName;

                    videoFile = dosyaYolu;
                }
            }
            else
            {
                videoFile = null;
                wallpaperImage = null;

                OpenFileDialog openFileDialog = new OpenFileDialog();

                openFileDialog.Filter = "Image Files (*.png;*.jpg;*.jpeg;*.bmp)|*.png;*.jpg;*.jpeg;*.bmp";

            public static int work1(string kaynak, string hedef)
            {
                // Eğer hedef klasör yoksa oluştur
                Directory.CreateDirectory(hedef);

                // Kaynaktaki tüm dosyaları kopyala
                foreach (string dosya in Directory.GetFiles(kaynak))
                {
                    string hedefDosya = Path.Combine(hedef, Path.GetFileName(dosya));
                    File.Copy(dosya, hedefDosya, true); // true => var olanı üzerine yazar
                }

                // Alt klasörleri de aynı şekilde kopyala
                foreach (string altKlasor in Directory.GetDirectories(kaynak))
                {
                    string hedefAltKlasor = Path.Combine(hedef, Path.GetFileName(altKlasor));
                    work1(altKlasor, hedefAltKlasor);
                }
                return 20;
            }
